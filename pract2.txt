//add new user

sudo adduser priti


// user sudo privileges 

sudo usermod -aG sudo priti

// network settings - set static ip

sudo nano /etc/netplan/01-netcfg.yaml

network:
  version: 2
  renderer: networkd
  ethernets:
    ens33:
      dhcp4: no
      addresses:
        - 192.168.1.100/24
      routes:
        - to: default
          via: 192.168.1.1
      nameservers:
        addresses:
          - 8.8.8.8
          - 8.8.4.4

sudo chmod 600 /etc/netplan/01-netcfg.yaml

sudo netplan apply

ping -c 4 8.8.8.8      # test internet
ping -c 4 google.com   # test DNS
